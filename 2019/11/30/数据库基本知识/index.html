数据库操作

1.显示

show databases;
默认数据库：

　　 mysql - 用户权限相关数据
　　 test - 用于用户测试数据
　　 information_schema - MySQL本身架构相关数据

2.创建

create database 数据库名称；

创建一个学校的数据库：
create database School;

3.删除

drop database 数据库名;

4.使用

use 数据库名 进入数据库

show tables; 查看表;

select database(); 查看当前使用的表

mysql 数据类型

1.数值

2.日期和时间类型

3.字符串类型

数据库表操作

1.创建表

  • 格式：
    CREATE TABLE table_name (column_name column_type [约束]);
    column_name 字段名
    column_type 字段数据类型
    约束可写可不写
  • 查看表：
    show tables; –>查看有哪些表
    desc student; –> 查看表字段的信息
    show create table student; –>查看表创建的信息
 //创建住宿人表
create table people(
    id int primary key,
    name char(10),
    id_rm int ,
    foreign key(id_rm) references room(id)
    on delete cascade #如果没有，删除时候不能先删被关联的，也不能先更改被关联的。
    on update cascade # 被关联的有更新或者删除都会关联，没有逗号
    );

//创建房间表
    create table room(
    id int primary key
    );

2.删除表

  • drop table 表名

3.修改表

 添加列：alter table 表名 add 列名 类型
删除列：alter table 表名 drop column 列名
修改列：
        alter table 表名 modify column 列名 类型;  -- 类型
        alter table 表名 change 原列名 新列名 类型; -- 列名，类型

添加主键：
        alter table 表名 add primary key(列名);
删除主键：
        alter table 表名 drop primary key;
        alter table 表名  modify  列名 int, drop primary key;

添加外键：alter table 从表 add constraint 外键名称（形如：FK_从表_主表） foreign key 从表(外键字段) references 主表(主键字段);
删除外键：alter table 表名 drop foreign key 外键名称

修改默认值：ALTER TABLE testalter_tbl ALTER i SET DEFAULT 1000;
删除默认值：ALTER TABLE testalter_tbl ALTER i DROP DEFAULT;

单表查询

表的基本信息


 学生表（学号，姓名，性别，年龄，院系)

 //学生表
create table student
    (
    sno char(9) primary key,
    sname char(20) unique,
    ssex char(2),

    sdept char(20)
    );
//课程表
create table course
    (
    cno char(4) primary key,
    cname char(40),
    cpno char(4),
    ccredit smallint,
    foreign key(cpno) references course(cno)

    );
    //成绩表
create table sc
    (
    sno char(9),
    cno char(4),
    grade smallint,
    primary key(sno,cno),
    foreign key(cno) references course(cno),
    foreign key(sno) references student(sno)
    on delete cascade
    on update cascade
    );

约束条件

null 与default

  • 含义：
    null: 表示可以为空
    not null :表示不可以为空
    default: 默认值
  • 理解：
    设置默认值后，则无论字段是null还是not null，都可以插入空，插入空默认填入default指定的默认值
  • 格式：
    字段 null 或者default

unique

  • 含义：
    设置位唯一不能重复
    可以设置复合唯一，单列可以重复，但不能同时重复
  • 格式：
    字段 unique
    unique(字段)
    unique(字段1，字段2,…..)

单列唯一

  • 注意：字段所对应的数据不能有重复
  • 插入数据:（1，”wang”,”优秀”)（1，”wg”,”良”)，出现错误id 重复
方法一：
create table department1(
id int unique,
name varchar(20) unique,
comment varchar(100)
);

方法二：
create table department(
        id int,
        name char(10),
        unique(id),
        unique(name)
    );

联合唯一

  • 注意：
    可以单列重复，不允许多列都重复
  • 数据：
    （1，”北京”,”192.168.57.35”,80)
    （1，”北京”,”192.168.57.35”,90)
    此时可以插入
    但是如果port 和host 都相等就会插入错误
 create table service(
        id int ,
        name varchar(20) unique,
        host varchar(15) not null,
        port int not null,
        unique(host,port)  #联合唯一
    );

primary key

  • 注意：
    一个表内只能有一个主键primary key
    =not null unique 不为空且唯一
    单列做主键：不为空且唯一；常用。
    多列做主键（复合主键）：多个字段连在一起，当作主键；
  • 格式：
    在某一个字段后用primary key
    在所有字段后单独定义primary key
在某一个字段后用primary key
create table department2(
id int primary key, #主键
name varchar(20),
comment varchar(100)
);


create table service(
ip varchar(15),
port char(5),
service_name varchar(10) not null,
primary key(ip,port)
);

auto_increment

  • 自增，默认每次增加一，添加在字段后面

foreign key

  • 格式：foreign key(dep_id) references dep(id)
#先建被关联的表,并且保证被关联的字段唯一
create table room(
    id int primary key
    );

#再建立关联的表
create table people(
    id int primary key,
    name char(10),
    id_rm int ,
    foreign key(id_rm) references room(id)
    on delete cascade #如果没有，删除时候不能先删被关联的，也不能先更改被关联的。
    on update cascade # 被关联的有更新或者删除都会关联，没有逗号
    );

#先往被关联表插入记录
insert into dep values

注意

如果在建有外键的关联表中，不设置on delete cascade会导致被关联表无法删除

delete from room where id =1;
select * from people;
1451 - Cannot delete or update a parent row: a foreign key constraint fails (`test`.`people`, CONSTRAINT `people_ibfk_1` FOREIGN KEY (`id_rm`) REFERENCES `room` (`id`))

总结

PRIMARY KEY (PK)    标识该字段为该表的主键，可以唯一的标识记录
FOREIGN KEY (FK)    标识该字段为该表的外键
NOT NULL    标识该字段不能为空
UNIQUE KEY (UK)    标识该字段的值是唯一的
AUTO_INCREMENT    标识该字段的值自动增长（整数类型，而且为主键）
DEFAULT    为该字段设置默认值
