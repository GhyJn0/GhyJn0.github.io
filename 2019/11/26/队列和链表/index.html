队列

  • 基本思想：
    队列是一种特殊的线性表，特殊之处在于只允许在表的前端进行删除操作，而在表的后端进行插入操作。和栈一样，队列是一种操作受限制的线性表。进行插入操作的端称为队尾，进行删除操作的端称为队头。
  • 操作：
    入队列（put）：只能在尾部进行入对操作。
    出队列（poll）：只能在头部进行出对操作
  • 特点：
    先进先出
  • 实现方式
    数组，
    链表，
    语言内置

数组实现队列

  • 注意

    • 判断队列满的条件
      rear==maxSize-1

    • 判断队列空的条件
      front==rear

class queue{
    private int maxSize;//数组最大长度
    private int front;//队列头
    private int  rear;//队列尾
    private int []arr;//数组
    //初始化队列基本信息
    public queue(int arrMaxSize){
        maxSize=arrMaxSize;//
        arr= new int[maxSize];
        front=-1;
        rear=-1;
    }
    //判断队列是否满了
    public boolean IsFull( ){
        return rear ==maxSize-1;
    }
    //判断队列是否空了
    public boolean IsEmpty(){
        return front==rear;
    }
    //添加数据
    public void addQueue(int data){
        if(!IsFull()){
            rear+=1;
            arr[rear]=data;
        }
        else{
            System.out.println("队列已经满了");
        }
    }
    //取数据
    public int getQueue(){
        if(!IsEmpty()){
            front+=1;
            return arr[front];
        }
        throw new RuntimeException("队列空，没有数据");
    }
    //取出所有的数据
    public void showQueue(){
        for(int i=0;i<arr.length;i++){
            System.out.printf("arr[%d]=%d",i,arr[i]);
        }
    }
    //取队首数据
    public int getFront(){
        if(!IsEmpty()){
            return arr[front+1];
        }
        throw new RuntimeException("队列为空");

    }
}

循环队列

  • 步骤
    • 1.front 就指向队列第一元素，初始值为0
    • 2.rear 指向队列最后一个元素，初始值为0
    • 3.为空时 front==rear
    • 4.满时 (rear+1)%maxsize==front
    • 5.元素的个数 (rear+maxsize-front)%maxsize
    • 6.删除一个元素 front=(front+1)%maxsize
    • 7.添加一个元素 rear=(rear+1)%maxsize
/*
 * 循环队列：
 * 1.front 就指向队列第一元素，初始值为0
 * 2.rear  指向队列最后一个元素，初始值为0
 * 3.为空时  front==rear
 * 4.满时   (rear+1)%maxsize==front
 * 5.元素的个数  (rear+maxsize-front)%maxsize
 * 6.删除一个元素 front=(front+1)%maxsize
 * 7.添加一个元素 rear=(rear+1)%maxsize
 */
class LoopQueue{
    private int maxSize;//数组最大长度
    private int front;//队列头
    private int  rear;//队列尾
    private int []arr;//数组
    //初始化队列基本信息
    public LoopQueue(int arrMaxSize){
        maxSize=arrMaxSize;//
        arr= new int[maxSize];
        front=0;
        rear=0;
    }
    public boolean IsFull( ){
        return (rear+1)%maxSize ==front;
    }
    //判断队列是否空了
    public boolean IsEmpty(){
        return front==rear;
    }
    //添加数据
    public void addQueue(int data){
        if(!IsFull()){
            arr[rear]=data;
            rear=(rear+1)%maxSize;

        }
        else{
            System.out.println("队列已经满了");
        }
    }
    //取数据
    public int getQueue(){
        if(!IsEmpty()){

            int data= arr[front];
            front=(front+1)%maxSize;
            return data;
        }
        throw new RuntimeException("队列空，没有数据");
    }
    //取出所有的数据
    public void showQueue(){
        for(int i=front;i<front+SumCount();i++){
            int value=i%maxSize;
            System.out.printf("arr[%d]=%d",value,arr[value]);
        }
    }
    public int SumCount(){
        return (rear+maxSize-front)%maxSize;
    }
    //取队首数据
    public int getFront(){
        if(!IsEmpty()){
            return arr[front];

        }
        throw new RuntimeException("队列为空");

    }

}

链表

  • 特点：

    • 链式存储结构的线性表将采用一组任意的存储单元存放线性表中的数据元素。由于不需要按顺序存储。
    • 1.链表是以节点的方式来存储
    • 2.每个节点包含data域《-》数据，next域《-》指向下一个节点
    • 3.链表中的各个节点不一定是连续的
  • 优点和缺点：
    链表在插入、删除数据元素时比顺序存储要快。
    查找一个节点时则要比顺序存储要慢

class LinkList{
    //初始化头节点
    private Link herohead =new Link(0,"");
    //获取头节点
    public Link getHerohead() {
        return herohead;
    }
        /*链表中随机添加节点
         * 1.找到当前链表最后节点
         * 2.将最后的节点指向这个新节点
         * */
    public void addLink(Link hero){
        //头节点不能动，赋值给他一个temp遍历
        Link temp=herohead;
        while(true){
            //找到最后一个节点
            if(temp.next==null){
                break;
            }
            //没找到让temp向后移动一个
            temp=temp.next;
        }
        temp.next=hero;
    }
    /*遍历链表*/
    public void listLink(){
        //判断链表是否为空
        if(herohead.next==null){
            System.out.println("链表空");
            return;
        }
        //头节点不能改，所以赋值给temp，让temp遍历
        Link temp=herohead.next;
        while(true){
            if(temp==null){
                break;
            }
            System.out.println(temp);
            temp=temp.next;
        }
    }
    /*删除链表*/
    public void  delete(int num){
        Link temp=herohead;
        Boolean flag=false;
        while (true){
          if(temp.next==null){
                break;
            }
          if(temp.next.no==num)
            {
                flag=true;
                break;
            }
            temp=temp.next;
        }
        if(flag){
            System.out.println("已经找到要删除的节点"+num);
            temp.next=temp.next.next;

        }
        else{

            System.out.println("没有找到找到要删除的节点");
        }

    }
    /*顺序建立链表*/
    public void add(Link hero){

        Boolean flag=false;
        //头节点赋值给temp
        Link temp=herohead;
        while(true){
            //判断是否到达尾部
            if(temp.next==null){
                break;
            }
            //升序排列
            else if(temp.next.no>hero.no){
                break;
            }
            //链表中已经存在节点
            else if(temp.next.no==hero.no){
               flag=true;
               break;
            }
            //temp往后移动一位
            temp=temp.next;
        }
        if(flag){
            System.out.println("数据已存在"+hero.no);
            return ;
        }
        hero.next=temp.next;
        temp.next=hero;
    }
    /*修改节点*/
    public void update(int num){
        //头节点赋值给temp
        Link temp= herohead;
        //根据flag判断是否存在
        Boolean flag=false;
        //遍历链表
        while(true){
            //找到编号
            if(temp.no==num){
                flag=true;
                break;
            }
            //遍历到尾节点
            else if(temp==null){
                break;
            }
            //temp往后移动一下
            temp=temp.next;
        }
        //flag 为true 说明找到了目标节点
        if (flag){
            String name=new Scanner(System.in).next();
            temp.name=name;
            return ;
        }
        System.out.println("没有找到编号");
    }
    /*求链表的长度*/
    public int length(){
        Link temp=herohead;
        int count=0;
        while(true){
            if (temp==null){
                return count-1;
            }
            temp=temp.next;
            count+=1;
        }
    }
    /*查找单表中倒数第k个节点
    * 1.先判断链表长L与K，小于K停止，大于K继续2
    * 2.遍历链表前 L-K个,或者用栈来遍历前K个
    */
    public Link search(int k){
        Link temp=herohead;
        int i=length()-k+1;
        System.out.println("i:"+i);
        if(i<=0){
            System.out.println("对不起，不存在");
            return null;
        }
        while(true){
            if (i>0){
                temp=temp.next;
            }
            else{
                System.out.println("数据："+temp);
                return temp;
            }
            i-=1;
        }

    }
    /*反转链表
    * 1.先定义一个reverhead=new Link()
    * 2.从头到尾遍历原来的链表，没遍历一个节点，就将其取出，放reversehead最前端
    * 3.原来的链表head.next=reversehead
    * */
    public void Reversal(Link herohead){
        if(herohead.next==null){
            return ;
        }
        //指向cur下一个节点
        Link next=null;
        //定义一个辅助指针，帮我们遍历原来的链表
        Link cur=herohead.next;
        //初始化一个reverse作为新链表节点
        Link Reversehead=new Link(0,"");

        while (cur!=null){
            next=cur.next;//暂时保存当前节点的下一个节点
            cur.next=Reversehead.next;//当前下一个节点指向新的链表最前端
            Reversehead.next=cur;//让cur连接到新的链表
            cur=next;//让cur向后移动一个节点
        }
        herohead.next=Reversehead.next;
    }
    /*逆序打印单链表
    * 1.用栈：先来后出*/
    public void ReverPrint(Link herohead){
        Link head= herohead;
        Stack <Link> Lstack=new Stack<Link>();
        while (head!=null){
            Lstack.add(head);
            head=head.next;
        }
        while(Lstack.size()>1){
            System.out.println(Lstack.pop());
        }

    }
}

双向链表

  • 定义：链表由单向的链变成了双向链。 使用这种数据结构，我们可以不再拘束于单链表的单向创建于遍历等操作，大大减少了在使用中存在的问题。

class DoubleLink{
    node head=new node(0,"");
    //头节点
    public node getHead() {
        return head;
    }

    //双向链表建立
    public void setLink(int id,String name){
        node temp=head;
        node ye=new node(id,name);
        while(temp.next!=null){
            temp=temp.next;
        }
        temp.next=ye;
        ye.pre=temp;
    }
    //双向链表的遍历
    public void listLink(){
        node temp=head.next;
        while(temp!=null){
            System.out.println(temp);
            temp=temp.next;
        }

    }
    //双向链表的删除
    public void deleteLink(int id){
        node temp=head.next;
        Boolean flag=true;
        while(true){
            if(temp==null)
            {
                break;
            }
            if(temp.id==id && temp.next!=null){
                temp.next.pre=temp.pre;
                temp.pre.next=temp.next;
                flag=false;
                break;
            }
            else if (temp.id==id && temp.next==null){
                temp.pre.next=null;
                flag=false;
                break;

            }
            temp=temp.next;
        }
        if (flag){
            System.out.println("对不起，你要删除的数据不存在");
        }
    }
    //双向链表的更新
    public void updateLink(int id){
        node temp=head;
        Boolean flag=true;
        while(temp!=null){
            if(temp.id==id){
                System.out.println("请输入更改的姓名");
                String nm=new Scanner(System.in).next();
                temp.name=nm;
                flag=false;
                break;
            }
            temp=temp.next;
        }
        if (flag){
            System.out.println("没有找到数据");
        }

    }
}

class Josefu{
    node first=null;
    node cur=null;
    //建立环形链表
    public void setJosefu( int num){
        for(int i=1;i<=num;i++){
            node newnode=new node(i);
            if(first==null){
                //新节点赋值给first
                first=newnode;
                //头节点的next指向自己
                first.next=first;
                //cur用来记录最后一个节点
                cur=first;
            }
            else{
                cur.next=newnode;
                newnode.next=first;
                cur=newnode;
            }

        }


    }
    //遍历环形链表
    public void listJosefu(){
        node temp=first;
        while(temp.next!=first){
            System.out.println("编号："+temp.id);
            temp=temp.next;
        }
        System.out.println("编号："+temp.id);
    }
    //每3次取出的编号
    public void getJosefu(){
        node temp=first;//头节点赋值给temp
        int count=0;
        int i=2;
        while(count!=10){//循环十次
            while(i<3) {//移动两次
                temp = temp.next;
                i++;
            }
            System.out.println("编号："+temp.next.id);//找到目标节点的上一个节点
            temp.next=temp.next.next;//删除目标节点
            count++;//自增
            i=1;//由于第一次循环temp=temp.next,temp没有算
        }
    }
}

稀疏数组

五子棋

这是一个五子棋的程序，有存盘退出和续上盘的功能，如下图，如何将下图的棋局进行保存呢？

那这个问题很简单，很多人可能会想到用二维数组来进行存储。

如上图，我们用0表示无子，1表示黑子，2表示蓝子，但是这个程序问题很大，因为该二维数组的很多值都是默认值0，因此记录了很多没有意义的数据，那么这个时候我们就可以使用稀疏数组来对该二维数组进行一个压缩。
那么稀疏数组到底是什么呢？
当一个数组中大部分元素是0，或者为同一个值的数组时，可以使用稀疏数组来保存该数组。

稀疏数组的处理方法是：
    记录数组一共有几行几列，有多少个不同的值
    把具有不同值的元素的行列以及值记录在一个小规模的数组中，从而缩小程序的规模

那么了解了稀疏数组的概念后，我们通过稀疏数组来改进一下五子棋程序。

经过稀疏数组的压缩之后，原数组从原来的11行11列变为了三行三列。
该稀疏数组的第一行记录的是原数组的行数和列数以及元素个数。
接下来的每一行记录的是有效元素的位置和值，例如第二行记录的是原数组中位于1，2位置上的元素1；第三行记录的是原数组中位于2,3位置上的元素2。

转换思路

  • 二维数组转稀疏数组的思路：

    • 遍历原始的二维数组，得到要保存的有效元素个数
    • 根据有效元素个数创建稀疏数组sparseArr
    • 将二维数组的有效数据存入稀疏数组即可

  • 稀疏数组转原始二维数组的思路：

    • 先读取稀疏数组的第一行，根据第一行的数据创建原始二维数组
    • 读取稀疏数组后几行的数据，并赋给原始的二维数组即可
public class SparseArray {
    public static void main(String[] args) {
        //数组---》稀疏数组
        int arr[][]=ArraytoSparse();
        //稀疏数组 ---》数组
        SparsetoArray(arr);

    }

    private static int [][] SparsetoArray(int [][] args) {
        //读取稀疏 数组的第一行0,0表示元数据的长度，0，1表示列数
        int Arr [][] =new int[args[0][0]][args[0][1]];
        //将稀疏数组转换为数组从第二行开始
        for(int i=1;i<=args[0][2];i++) {
            Arr[args[i][0]][args[i][1]] = args[i][2];
        }
        //遍历数组
        for (int []row:Arr){
            for (int data:row){
                System.out.printf("%d\t",data);
            }
            System.out.println();
        }
        //返回稀疏数组转换后的数组
        return Arr;

    }

    private static int [][] ArraytoSparse() {
        //创建一个原始的二维数组11*11 0：表示没有棋子，1：表示黑子，2：表示篮子
        int chessArr[][]=new int[11][11];
        chessArr[1][2]=1;
        chessArr[2][4]=2;
        //输出原始数组
        for(int[] row: chessArr)
        {
            for(int data: row){
                System.out.printf("%d\t",data);
            }
            System.out.println();
        }
        //遍历二维数组，得到非0数据的个数
        int sum=0;
        for(int[] row: chessArr)
        {
            for(int data: row){
                if(data!=0)
                {
                    sum++;
                }
            }

        }
        //2.创建对应的二维数组
        int sparseArr[][] =new int [sum+1][3];
        //给稀疏数组赋值
        sparseArr[0][0]=11;
        sparseArr[0][1]=11;
        sparseArr[0][2]=sum;
        int count=1;
        //3.给稀疏数组赋值
        for(int i=0;i<11;i++)
        {
            for (int j=0;j<11;j++)
            {
                if(chessArr[i][j]!=0){
                    sparseArr[count][0]=i;
                    sparseArr[count][1]=j;
                    sparseArr[count][2]=chessArr[i][j];
                    count++;
                }
            }
        }
        //4.输出稀疏数组
        System.out.println("稀疏数组长度："+sparseArr.length);
        for(int [] r: sparseArr)
        {
            for (int data:r){
                System.out.printf("%d\t",data);
            }
            System.out.println();
        }
        //返回稀疏数组
        return sparseArr;

    }
