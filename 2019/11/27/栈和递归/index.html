栈和递归

君の名は · 2019-11-27 · 次阅读

栈

栈的概念和基本操作

  • 概念： 
          栈顶：允许插入和删除的这一端称为栈顶。
  栈底：另外固定的这一端称为栈底。
  空栈：不含任何元素的栈称为空栈。
  • 特点：先进后出

栈代码

class ArrayStack{
    public int maxSize;//栈的大小
    private int [] stack;//栈
    public int top=-1;
    //构造器
    public ArrayStack(int maxSize) {
        this.maxSize = maxSize;
        stack= new int[ this.maxSize];
    }
    //判断栈是否为空
    public Boolean empty(){
        return top==-1;
    }
    //判断栈是否满
    public Boolean full(){
        return top==maxSize-1;
    }
    //把数据压入栈中
    public void push(int num){
        if(full()){
            System.out.println("栈满了");
            return;
        }
        top=top+1;
        stack[top]=num;
    }
    //取出数据
    public int pop(){
        if(empty()){
            throw new RuntimeException("栈空了");
        }
        int data=stack[top];
        top--;
        return data;
    }
    //遍历栈
    public void list(){
        while(top>=0){
            System.out.println(stack[top]);
            top--;
        }
    }

}

测试代码

public class Cetry {
    public static void main(String[] args) {
        ArrayStack obj=new ArrayStack(5);

        obj.push(1);
        obj.push(2);
        obj.push(3);
        obj.push(4);
        obj.push(5);
        System.out.println("取出栈数据："+obj.pop());
        System.out.println("遍历栈");
        obj.list();

    }
}

计算器

  • 步骤：
    • 1.声明一个数字的栈，一个符号的栈
    • 2.遍历字符串，数字添加到数字栈中，符号添加符号栈
    • 3.如果符号栈为空直接添加进去，如果符号栈已经有符号，需要比较新的符号和栈顶的符号的优先级
    • 4 * / 》 + -
    • 5.如果新的符号的优先级大于栈顶的优先级，把新的符号添加到符号栈顶
    • 否则符号栈pop栈顶，数字栈pop两个数据运算结果，数据栈加入结果，符号加入符号栈

计算器代码

class Calculator{
    public int maxSize;//栈的大小
    private int [] stack;//栈
    public int top=-1;
    //构造器
    public Calculator(int maxSize) {
        this.maxSize = maxSize;
        stack= new int[ this.maxSize];
    }
    //判断栈是否为空
    public Boolean empty(){
        return top==-1;
    }
    //判断栈是否满
    public Boolean full(){
        return top==maxSize-1;
    }
    //获取栈顶
    public int getTop(){
        return stack[top];
    }

    //把数据压入栈中
    public void push(int num){
        if(full()){
            System.out.println("栈满了");
            return;
        }
        top=top+1;
        stack[top]=num;
    }
    //取出数据
    public int pop(){
        if(empty()){
            throw new RuntimeException("栈空了");
        }
        int data=stack[top];
        top--;
        return data;
    }
    //遍历栈
    public void list(){
        while(top>=0){
            System.out.println(stack[top]);
            top--;
        }
    }
    // 判断是否是运算符
    public Boolean Notnum(char a){
        return a=='+'|| a=='-'|| a=='*'|| a=='/';
    }

    //计算结果
    public int count(int num1,int num2,int a){
        int data=0;
        switch (a){
            case '+':
                data=num1+num2;
                System.out.printf("计算：%d+%d=%d",num1,num2,data);
                break;
            case '-':
                data=num2-num1;
                System.out.printf("计算：%d-%d=%d",num2,num1,data);
                break;
            case '*':
                data=num1* num2;
                System.out.printf("计算：%d*%d=%d",num1,num2,data);
                break;
            case '/':
                data=num2/num1;
                System.out.printf("计算：%d/%d=%d",num2,num1,data);
                break;
            default:
                break;
        }

        return data;
    }
    //优先级
    public int priority(int opr){
        if(opr=='+'|| opr =='-'){
            return 1;
        }
        else if( opr=='*' || opr=='/'){
            return 2;
        }
        else{
            return -1;
        }
    }

}

测试代码

public class Computer {
    public static void main(String[] args) {
        String posture ="70+2*4-8";

        //初始化一个符号栈
        Calculator mark =new Calculator(6);
        //初始化一个数据栈
        Calculator num =new Calculator(7);
        int index=0;//初始化一个索引地址
        int pre=0;
        int num1=0;
        int num2=0;
        int re=0;
        char ch=' ';
        String keepnum="";
        int data=0;
        while(true){
            ch=posture.substring(index,index+1).charAt(0);//获取每一个字符
            //如果是运算符
            if(mark.Notnum(ch) ){
                //如果符号栈为空
                if (mark.empty()){
                        mark.push(ch);
                }
                else{
                    //如果新遍历的符号小于等于栈顶的优先级，
                    if(mark.priority(ch)<=mark.priority(mark.getTop())){
                        pre=mark.pop();
                        num1=num.pop();
                        num2=num.pop();
                        re=num.count(num1,num2,pre);
                        //System.out.println("计算的值："+pre);
                        num.push(re);
                        mark.push(ch);
                    }
                    else{
                        mark.push(ch);
                    }
                    //运算符栈不为空
                }

            }
            else{
                /*处理多位数
                * 1.当处理多位数时，不能发下一个整数立即入栈
                * 2.处理时，多判断一位，看这一位是数字还是符号，如果是符号，就停止
                * 不是符号用一个字符串加起来
                * */
                System.out.println("\n添加数字是:"+ch);
                keepnum+=ch;
                //如果是最后一位时，就不用判断

                if(index==posture.length()-1){
                  data=Integer.parseInt(keepnum);
                  num.push(data);
                }
                else {
                    if(num.Notnum(posture.substring(index+1,index+2).charAt(0))) {
                        data=Integer.parseInt(keepnum);
                        num.push(data);
                        keepnum="";
                    }
                }


            }
            index++;
            if(index>=posture.length()){
                break;
            }
        }
        while (true){
            if(mark.empty()){
                break;
            }
            num1=num.pop();
            num2=num.pop();
            pre=mark.pop();
            re=num.count(num1,num2,pre);
            num.push(re);
        }
        System.out.printf("\n%s= %d",posture,num.pop());

    }
}

添加数字是:2
添加数字是:4

计算：4*2=8添加数字是:8

计算：8-8=0
计算：0+70=70
70+2*4-8= 70
