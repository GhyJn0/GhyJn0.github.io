xpath 选择器

使用


import requests
from lxml import etree

response=requests.get("http://www.bilibili.com/")
response.encoding="utf-8"
tree=etree.HTML(response.text)
span_list=tree.xpath("//div[@id='primaryChannelMenu']/span")
text=[]
for span in span_list:
    text.append(span.xpath("./div[2]/a/span/text()")[0])
print(text)
# 结果

语法

# 选取节点
//节点    从匹配选择的当前节点选择文档中的节点，而不考虑它们的位置
//div   选取所有 div子元素 而不管它们在文档中的位置
.      获取当前节点

..      获取父亲节点

@       获取属性
<a href="//www.bilibili.com/v/music/" class="name">
//a/@href   获取的是://www.bilibili.com/v/music/
    
# 谓语选择
//节点[@属性='']  选取特定的div 
//div[@id='primaryChannelMenu']

//节点1/节点[1]   获取节点1下第一个节点
//div[@id='primaryChannelMenu']/span/div[2]/a/span/text()


# 获取文本
