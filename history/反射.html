获取构造方法

Constructor<?>[] getConstructors()返回所有公共构造方法对象的数组

Constructor<?>[] getDeclaredConstructors() 返回所有构造方法对象的数组

Constructor getConstructor(Class<?>... parameterTypes) 返回单个公共构造方法对象

Constructor getDeclaredConstructor(Class<?>...parameterTypes)返回单个构造方法对象

    
T newInstance(Object...initargs) 根据指定的构造方法创建对象

 /**
         * getConstructors()
         * return  返回一个对象数组，表示类的所有公共构造函数
         */
        Class<Student> c = Student.class;
        Constructor<?>[] cons = c.getConstructors();
        for (Constructor<?> con : cons) &#123;
            System.out.println(con);
        &#125;

 /**
         *
         * getDeclaredConstructors()
         * return   返回一个对象数组，表示类的所有构造函数包括私有和公有
         */

        Constructor<?>[] cons1 = c.getDeclaredConstructors();
        for (Constructor<?> con : cons1) &#123;
            System.out.println(con);
        &#125;

  /**
         *
         * getConstructor( 参数)：
         * 参数：
         *      获取的构造方法的参数个数和数据类型对应的字节码文件对象
         *
         * return   返回一个对象，表示类的指定的构造方法
         */
        Constructor<Student> con2 = c.getConstructor(int.class,String.class);
        Student obj1 = con2.newInstance(19,"xiaobai");
        System.out.println(obj1);  
        //相当于下面
        Student s=new Student(19,"xiaobai");
         System.out.println(s);

 /**
         *
         * getDeclaredConstructor(参数)
         * 参数：
         *      获取的构造方法的参数个数和数据类型对应的字节码文件对象

         * return   返回一个对象，表示类的所有构造函数包括私有和公有
         */
        Constructor<Student> con3 = c.getDeclaredConstructor(int.class);
        //暴力反射 获取私有的构造方法
        con3.setAccessible(true);
        Student obj2 = con3.newInstance(19);
        System.out.println(obj2);

获取成员变量

Field[] getFields() 返回所有公共成员变量对象的数组

Field[] getDeclaredFields() 返回所有成员变量对象的数组

Field getField(String name) 返回单个公共成员变量对象

Field getDeclaredField(String name) 返回单个成员变量对象

 void set(Object obj,Object value) 给obj对象的成员变量赋值为value

  /**
         * getFields()返回一个包含field对象的数组       ======公有成员变量
         */
        Field[] fields = c.getFields();
        for (Field field : fields) &#123;
            System.out.println(field);

        &#125;

  /**
         * getDeclareFields() 返回一个包含field对象的数组  ======所有成员变量 私有或者公有
         */
        Field[] fields1 = c.getDeclaredFields();
        for (Field field : fields1) &#123;
            System.out.println(field);

        &#125;

 /**
         * getField( String name)
         参数：成员变量的名字
         * 返回一个======指定的公共成员变量
         */
        Field address = c.getField("address");
        System.out.println(address);

   /**
         *  getDeclareField(String name)
         *  放回一个======指定的私有或者公有的成员变量
         */
        Field age = c.getDeclaredField("age");
        System.out.println(age);

        //1.获取studentclass对象
        Class<Student> c = Student.class;
        //2.创建无参构造函数
        Constructor<Student> cons = c.getConstructor();
        Student obj = cons.newInstance();
        System.out.println(obj);
        //3.获取成员变量
        Field nameField = c.getDeclaredField("name");
        Field ageField = c.getDeclaredField("age");
        Field addressField = c.getDeclaredField("address");
        //4.为成员变量赋值
        //私有成员变量赋值前暴力反射
        nameField.setAccessible(true);
        nameField.set(obj,"小白");
        ageField.setAccessible(true);
        ageField.set(obj,19);
        //公有成员变量赋值前不需要暴力反射
        addressField.set(obj,"南京");
        System.out.println(obj);
        /*Student:name:null            age:0            address:null
        Student:name:小白            age:19            address:南京*/

 public static void method() throws Exception&#123;
        //1.创建class对象
        Class<teacher> tClass = teacher.class;

        //2.创建无参构造对象 Teacher con=new Teacher();
        Constructor<teacher> con = tClass.getConstructor();
        teacher obj = con.newInstance();

        //3.调用method1 方法 con.method1()    public void method1()
        Method method1Fun = tClass.getMethod("method1");
        method1Fun.invoke(obj);

        //4.调用method2 方法  public void method2(String s)
        Method method2Func = tClass.getMethod("method2", String.class);
        method2Func.invoke(obj,"参数");

        //5.调用method3 方法   public String method3(String s, int i)
        Method method3Func = tClass.getMethod("method3", String.class, int.class);
        Object str = method3Func.invoke(obj, "小白", 19);
        System.out.println(str);

        //6.调用function 方法  private void function()
        Method functionFunc = tClass.getDeclaredMethod("function");
        //设置暴力反射
        functionFunc.setAccessible(true);
        functionFunc.invoke(obj);
