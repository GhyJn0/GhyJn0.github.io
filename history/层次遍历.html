# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def levelOrder(self, root: TreeNode) -> List[List[int]]:
        if root==None:
            return []
        queue=[root]
        listList=[]
        while queue!=[]:
            lis=[]
            length=len(queue)
            for i in range(length):
                node=queue.pop(0)
                lis.append(node.val)
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)

            listList.append(lis)
        return listList
